0. Installation

https://github.com/filipezabala/voice

Load packages and audio files

# packs
library(voice)

# get path to audio file
path2wav <- list.files(system.file('extdata', package = 'wrassp'),
                       pattern <- glob2rx('*.wav'), full.names = TRUE)
