3.2. Music sheet

Compress to 1%

# compress
M1 <- voice::interp_df(M, 0.01, id = 3)

# assign notes
M1$f0_spn <- voice::notes(M1$f0)

# duration
d1 <- voice::duration(M1$f0_spn)

# gm by Renfei Mao
library(gm)
m <- gm::Music()
m <- m +
  gm::Meter(4, 4) +
  gm::Line(pitches = as.list(as.character(d1$note)),
           durations = as.list(d1$dur_line)) +
  gm::Tempo(170)
gm::show(m, to = c('score', 'audio'))
